// ================= SETUP =================
void setup() {
  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, LOW);

  pinMode(LED_RED, OUTPUT);
  pinMode(LED_YELLOW, OUTPUT);
  pinMode(LED_GREEN, OUTPUT);

  digitalWrite(LED_YELLOW, LOW);
  digitalWrite(LED_GREEN, LOW);

  irrecv.enableIRIn();
}

// ================= LOOP =================
void loop() {

  // ---- IR ----
  if (irrecv.decode(&results)) {
    handleIR(results.value);
    irrecv.resume();
  }

  // ---- CLAP ----
  if (currentMode == MODE_CLAP) {
    int mic = analogRead(MIC_PIN);
    micAvg = micAvg * 0.95 + mic * 0.05;

    int diff = mic - micAvg;
    if (diff > clapDelta && millis() - clapTimer > 150) {
      clapTimer = millis();
      clapCount++;
    }

    if (clapCount == 2 && millis() - clapTimer < 60) {
      digitalWrite(RELAY_PIN, !digitalRead(RELAY_PIN));
      clapCount = 0;
    }

    if (millis() - clapTimer > 700) clapCount = 0;
  }

  // ---- LDR ----
  if (currentMode == MODE_LDR) {
    int raw = analogRead(LDR_PIN);
    ldrSmooth = ldrSmooth * 0.9 + raw * 0.1;

    if (ldrSmooth < ldrThreshold) digitalWrite(RELAY_PIN, HIGH);
    else digitalWrite(RELAY_PIN, LOW);
  }
}

// ================= IR HANDLER =================
void handleIR(unsigned long code) {
  switch (code) {


    case IR_ON:
      currentMode = MODE_ON;
      digitalWrite(RELAY_PIN, HIGH);
      break;

    case IR_OFF:
      currentMode = MODE_OFF;
      digitalWrite(RELAY_PIN, LOW);
      break;

    case IR_LDR:
      currentMode = MODE_LDR;
      break;

    case IR_CLAP:
      currentMode = MODE_CLAP;
      break;

    case IR_NEXT:
      currentMode = (Mode)((currentMode + 1) % 4);
      break;

    case IR_PREV:
      currentMode = (Mode)((currentMode + 3) % 4);
      break;
  }
}
